{
	//æ£€æŸ¥ä¼ å…¥çš„å¯¹è±¡
	if(SourceActor == nullptr||TargetActor == nullptr||GameplayEffect == nullptr)

}

{

	if (!LocalPlayer)

	if (!PlayerState)
	{
		return;
	}

	// è·å–å±æ€§
	ExorcistAttributeSet = CastChecked<UExorcistAttributeSet>(PlayerState->GetAttributeSet());
	if (!ExorcistAttributeSet)
	{

	}
}
